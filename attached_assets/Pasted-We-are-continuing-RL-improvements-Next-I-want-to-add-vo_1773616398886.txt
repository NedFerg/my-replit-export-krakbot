# Base volume proportional to volatility (more vol → more activity)
base_vol = max(1.0, abs(price - prev_price) * 1000)

# Random microstructure noise

# Split into buy/sell volume using price movement direction
if price > prev_price:
    buy_vol = volume * random.uniform(0.55, 0.75)
    sell_vol = volume - buy_vol
elif price < prev_price:
    sell_vol = volume * random.uniform(0.55, 0.75)
    buy_vol = volume - sell_vol
else:
    # Flat price → balanced flow
    buy_vol = volume * 0.5
    sell_vol = volume * 0.5

# Append to histories
self.volume_history.append(volume)
self.buy_volume_history.append(buy_vol)
self.sell_volume_history.append(sell_vol)

3. Compute rolling volume + imbalance + pressure

Still inside the step loop, after updating the histories:
python

# Rolling total volume
if len(self.volume_history) >= self.volume_window:
    vol_slice = self.volume_history[-self.volume_window:]
    rolling_vol = sum(vol_slice) / len(vol_slice)
else:
    rolling_vol = 0.0

# Volume imbalance: buy - sell
vol_imbalance = buy_vol - sell_vol

# Buy/sell pressure: normalized imbalance
denom = buy_vol + sell_vol
if denom > 0:
    pressure = vol_imbalance / denom
else:
    pressure = 0.0

# Attach to state
state.rolling_vol = rolling_vol
state.vol_imbalance = vol_imbalance
state.pressure = pressure
