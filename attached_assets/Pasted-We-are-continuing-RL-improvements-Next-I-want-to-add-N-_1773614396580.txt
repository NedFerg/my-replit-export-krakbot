def add_experience(self, prev_state, action, reward, new_state, done):
    # Push raw 1-step transition into N-step buffer
    self.n_step_buffer.append((prev_state, action, reward, new_state, done))

    # If we don't yet have N steps and episode not done, wait
    if len(self.n_step_buffer) < self.n_step and not done:
        return

    # Build N-step return from buffer
    R = 0.0
    gamma_pow = 1.0
    for (_, _, r, _, d) in self.n_step_buffer:
        R += gamma_pow * r
        if d:
            break
        gamma_pow *= self.gamma

    s0, a0, _, _, _ = self.n_step_buffer[0]
    s_n = self.n_step_buffer[-1][3]

    n_step_transition = (s0, a0, R, s_n)

    # Prioritized replay: assign initial max priority
    if self.replay_buffer:
        max_prio = max(p for (_, p) in self.replay_buffer)
    else:
        max_prio = 1.0

    self.replay_buffer.append((n_step_transition, max_prio))

    # Maintain capacity as before
    if len(self.replay_buffer) > self.replay_capacity:
        self.replay_buffer.pop(0)

    # If episode ended, clear N-step buffer
    if done:
        self.n_step_buffer.clear()
