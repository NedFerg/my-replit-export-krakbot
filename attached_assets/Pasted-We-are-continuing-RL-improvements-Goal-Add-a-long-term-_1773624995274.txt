# --- Long-term cycle scoring ---
btc_hist = self.asset_prices["BTC"]

if len(btc_hist) >= 200:
    # 1. Long-horizon BTC trend (e.g. 200-step)
    ret_200 = (btc_hist[-1] - btc_hist[-200]) / max(btc_hist[-200], 1e-6)

    # 2. Long-horizon realized vol (approx)
    window = btc_hist[-200:]
    mean_p = sum(window) / len(window)
    var_p = sum((p - mean_p) ** 2 for p in window) / len(window)
    vol_200 = (var_p ** 0.5) / max(mean_p, 1e-6)

    # 3. Average altseason over long window (use stored history if available, else approximate)
    alt_long = self.altseason_index  # placeholder; agent can refine with a buffer

    score = 0.0

    # Strong uptrend → later in cycle
    if ret_200 > 0.5:
        score += 1.0
    if ret_200 > 1.0:
        score += 1.0

    # High long-term vol → late / post-crash
    if vol_200 > 0.10:
        score += 1.0

    # Strong altseason → mid/late
    if alt_long > 0.5:
        score += 0.5
    if alt_long > 0.8:
        score += 0.5

    self.cycle_score = score

    # Map score → phase
    if score < 0.5:
        self.cycle_phase = 0   # early accumulation
    elif score < 1.5:
        self.cycle_phase = 1   # mid expansion
    elif score < 2.5:
        self.cycle_phase = 2   # late distribution
    else:
        self.cycle_phase = 3   # post-crash / reset
else:
    # Not enough history yet
    self.cycle_score = 0.0
    self.cycle_phase = 0
