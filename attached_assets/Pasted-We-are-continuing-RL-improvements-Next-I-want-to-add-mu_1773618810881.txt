2. Generate synthetic prices for all assets each step

Inside the main simulation loop (where SOL price is updated), add:

# --- Multi-asset synthetic price updates ---
for a in self.assets:
    if a == "SOL":
        continue  # SOL already updated by main logic

    prev_price = self.asset_prices[a][-1] if self.asset_prices[a] else 100.0

    # correlated movement: BTC leads, ETH follows, others react
    if a == "BTC":
        drift = random.uniform(-0.002, 0.002)
    elif a == "ETH":
        drift = random.uniform(-0.003, 0.003) + 0.3 * (self.asset_prices["BTC"][-1] - prev_price) / prev_price
    else:
        drift = random.uniform(-0.004, 0.004) + 0.2 * (self.asset_prices["ETH"][-1] - prev_price) / prev_price

    new_price = max(0.1, prev_price * (1 + drift))
    self.asset_prices[a].append(new_price)

This creates realistic cross‑asset correlation structure.

4. Compute volume + order‑flow features for each asset

Add:
python

for a in self.assets:
    prev_price = self.asset_prices[a][-2] if len(self.asset_prices[a]) >= 2 else self.asset_prices[a][-1]

    # synthetic volume
    base_vol = max(1.0, abs(self.asset_prices[a][-1] - prev_price) * 1000)
    noise = random.uniform(0.8, 1.2)
    volume = base_vol * noise

    if self.asset_prices[a][-1] > prev_price:
        buy_vol = volume * random.uniform(0.55, 0.75)
        sell_vol = volume - buy_vol
    elif self.asset_prices[a][-1] < prev_price:
        sell_vol = volume * random.uniform(0.55, 0.75)
        buy_vol = volume - sell_vol
    else:
        buy_vol = sell_vol = volume * 0.5

    self.asset_volume_history[a].append(volume)
    self.asset_buy_volume_history[a].append(buy_vol)
    self.asset_sell_volume_history[a].append(sell_vol)

    # rolling volume
    if len(self.asset_volume_history[a]) >= 20:
        self.asset_rolling_vol[a] = sum(self.asset_volume_history[a][-20:]) / 20
    else:
        self.asset_rolling_vol[a] = 0.0

    # imbalance + pressure
    imb = buy_vol - sell_vol
    denom = buy_vol + sell_vol
    pressure = imb / denom if denom > 0 else 0.0

    self.asset_vol_imbalance[a] = imb
    self.asset_pressure[a] = pressure
