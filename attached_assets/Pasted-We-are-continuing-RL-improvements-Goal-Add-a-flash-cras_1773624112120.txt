# --- Flash-crash / panic detection ---
vol = self.vol_regime
liq = self.liq_regime
alt = self.altseason_index

# 1. Sudden volatility spike (fast-moving)
vol_spike = (vol - self.prev_vol_regime) > 0.015

# 2. Liquidity collapse
liq_crash = liq < 0.2

# 3. Sharp price drop across multiple assets
drop_count = 0
for a in self.assets:
    if len(self.asset_prices[a]) > 2:
        r = (self.asset_prices[a][-1] - self.asset_prices[a][-2]) / max(self.asset_prices[a][-2], 1e-6)
        if r < -0.03:  # 3% drop in one step
            drop_count += 1

broad_selloff = drop_count >= 3  # at least 3 assets nuking simultaneously

# Score panic conditions
panic_score = 0
if vol_spike: panic_score += 1
if liq_crash: panic_score += 1
if broad_selloff: panic_score += 2

# Convert to panic_risk
if panic_score >= 3:
    self.panic_risk = 2
elif panic_score >= 1:
    self.panic_risk = 1
else:
    self.panic_risk = 0
